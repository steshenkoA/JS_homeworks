function pow(numberResult, degreeResult) {
    if (degreeResult == 0) {
        return 1;
    } else if (degreeResult == 1) {
        return numberResult;
    } else {
        let result = 1;
        if (degreeResult > 1) {
            for (let i = 0; i < degreeResult; i++) {
                result *= numberResult;
            }
        }
        return result;
    }
}
