function findValueByKey(companyName, obj = company) {
  if (obj.name === companyName && obj.type === 'subCompany') {
    let { partners, ...clientInfo } = obj;
    return clientInfo;
  }
let value = ['clients', 'partners'];
  for (let key of value) {
    if (obj[key]) {
      for (let item of obj[key]) {
        let result = findValueByKey(companyName, item);
        if (result) return result;
      }
    }
  }

  return null;
}
