var services = {
    "стрижка": "60 грн",
    "гоління": "80 грн",
    "Миття голови": "100 грн"
};
services["Розбити скло"] = "200 грн";

function price(sum) {
    return sum;
}

function minPrice(minPrice) {
    return minPrice;
}

function maxPrice(maxPrice) {
    return maxPrice;
}

var priceServices = {};
var sum = 0;
var min = Infinity; 
var max = -Infinity; 

for (var key in services) {
    priceServices[key] = services[key].split(' ')[0];
    var numberValue = Number(priceServices[key]);
    sum += numberValue;
    
    if (numberValue < min) {
        min = numberValue;
    }
    if (numberValue > max) {
        max = numberValue;
    }
}

console.log("Сума: " + price(sum) + ", Мінімальна ціна: " +  minPrice(min) + ", Максимальна ціна:  " +   maxPrice(max));
